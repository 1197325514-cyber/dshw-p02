1.3.2 下载函数

def download_stock_data(stock_code, start_date, end_date):
    """下载单只股票的后复权日度行情数据"""
    df = ak.stock_zh_a_hist(
        symbol=stock_code,
        period="daily",
        start_date=start_date,
        end_date=end_date,
        adjust="hfq"  # 后复权
    )
    return df
